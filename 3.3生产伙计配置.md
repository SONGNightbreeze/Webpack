# 7. 生产环境基本配置
# webpack.config.js
```js
const {resolve} = require('path');
// 引入将css从js中分离的插件
const MiniCssExtractPlugin = require('mini-css-extract-plugin');

// browserslist 默认使用的生产环境配置，如果想使用开发环境配置
// 定义nodejs的环境变量：决定使用browserslist的哪个环境
process.env.NODE_ENV = 'production'  // 'development'

module.exports = {
    entry:'./src/js/index.js',
    output:{
        filename:'js/built.js',
        path:resolve(__dirname,'build')
    },
    module:{
        rules:[
            {
                test:/\.css$/,
                use:[
                    // 所以要将'style-loader'替换成MiniCssExtractPlugin
                    // 'styles-loader',
                    MiniCssExtractPlugin.loader, 
                    'css-loader'
                    // 现在需要运用插件将css从js中分离出来

                    // 对样式做兼容性处理
                    {
                        // 还需要在package.json中定义browserslist
                        loader:'postcss-loader',
                        // 通过options修改默认配置
                        options:{
                            ident:'postcss',
                            // 只是postcss以什么插件进行工作
                            plugins:() => [
                                require('postcss-preset-env')()
                            ]
                        }
                    }
                ]
                // 现在就能处理css文件还能将css从js里提取成单独文件
            },
            {
                test:/\.less$/,
                use:[
                    MiniCssExtractPlugin.loader, 
                    'css-loader',
                    // 将兼容性处理的配置放到'css-loader'和'less-loader'中间
                    // 因为use的数组的执行顺序是从下向上执行
                    {
                        loader:'postcss-loader',
                        options:{
                            ident:'postcss',
                            plugins:() => [
                                require('postcss-preset-env')()
                            ]
                        }
                    },
                    'less-loader'
                ]
            },
        ]
    },
    plugins:[
        new MiniCssExtractPlugin({
            // 将输出的文件再不同的路径下
            filename:'css/built.css'
        })
    ],
    mode:'production'
}

```

# 因为处理css文件兼容性的配置重复，可以在上面定义一个变量
```js
const {resolve} = require('path');
const MiniCssExtractPlugin = require('mini-css-extract-plugin');
// 对css进行压缩插件
const  OptimizeCssAssetsWebpackPlugin = require('optimize-css-assets-webpack-plugin')
// 处理html文件
const HtmlWebpackPlugin = require('html-webpack-plugin');

process.env.NODE_ENV = 'production' 

// 复用loader
const commonCssLoader = [
    MiniCssExtractPlugin.loader,
    'css-loader',
    {
        loader:'postcss-loader',
            options:{
                ident:'postcss',
                plugins:() => [
                    require('postcss-preset-env')()
                ]
            }
    }
]; 

module.exports = {
    entry:'./src/js/index.js',
    output:{
        filename:'js/built.js',
        path:resolve(__dirname,'build')
    },
    module:{
        rules:[
            {
                test:/\.css$/,
                // 利用扩展运算符 ... 就可以将他展开
                use:[...commonCssLoader]

            },
            {
                test:/\.less$/,
                use:[...commonCssLoader,'less-loader']
            },
            // 处理js资源，用eslint做语法检查
            // 正常来将，一个文件只能被一个loader处理
            // 当一个文件要被多个loader处理，那么一定要指定loader执行的先后顺序
            // 先执行eslint，在执行babel  可以添加 enforce:'pre'
            {
                // 在package.json中要添加eslintConfig --> airbnb
                test:/\.js$/,
                exclude:/node_modules/,
                // 优先执行
                enforce:'pre',
                loader:'eslint-loader',
                // 将检测的错误自动修复
                options:{
                    fix:true
                }
            },
            // js的兼容性处理
            {   
                test:/\.js$/,
                exclude:/node_modules/,
                loader:'babel-loader',
                options:{
                    presets:[
                        // 基本的js兼容性处理
                        //'@babel/preset-env',
                        [
                            '@babel/preset-env',
                            // 按需加载 
                            {
                                useBuiltInt:'usage',
                                corejs:{version:3},
                                // 兼容哪些版本的浏览器
                                targets:{
                                    chrome:'60',
                                    firefox:'50'
                                }
                            }
                        ]
                    ]
                }
            },
            // 处理图片
            {
                test:/\.(jpg|png|gif)/,
                loader:'url-loader',
                options:{
                    limit:8 * 1024,
                    name:'[hash:10].[ext]',
                    //输出的路径
                    outputPath:'imgs',
                    esModule:false
                }
            },
            // 处理html里的图片
            {
                test:/\.html$/,
                loader:'html-loader', 
                // 注意这里使用的是commonjs，而url引入图片使用的是es6
                // 所以要在上面的配置里面关掉es6      esModule:false
            },
            // 处理其他文件 
            {
                exclude:/\.(js|css|less|html|jpg|png|gif)/,
                // 'file-loader'会原封不动的输出文件
                loader:'file-loader',
                options:{
                    //让其他文件都输出到media同一个文件夹下
                    outputPath: 'media'
                }
            }
        ]
    },
    plugins:[
        new MiniCssExtractPlugin({
            filename:'css/built.css'
        }),
        // 调用压缩css的插件
        new OptimizeCssAssetsWebpackPlugin(),
        new HtmlWebpackPlugin({
            template:'./src/index.html',
            // 压缩html
            minify:{
                collapseWhitespace:true,
                removeComments:true
            }
        })
    ],
    // mode为'production' js就自动压缩
    mode:'production'
}
```
